#include <iostream>

struct Node
{
    int value;
    Node* left = nullptr;
    Node* right = nullptr;

    Node()
        : value(0) { }

    Node(int val)
        : value(val) { }
};

Node* findNode(Node* root, int val)
{
    if (root == nullptr)
    {
        return nullptr;
    }

    if (val < root->value)
    {
        return findNode(root->left, val);
    }
    else if (val > root->value)
    {
        return findNode(root->right, val);
    }
    
    return root;
}

int main()
{
    Node root(5), left(4), right(10);
    root.left = &left;
    root.right = &right;

    Node* node = findNode(&root, 10);
    std::cout << node;

    return 0;
}
